# Campbell Family Hub - Complete Implementation Specification

## Executive Summary

Build a complete family property management and collaboration platform. This is a white-label-ready system that will serve the Campbell family immediately (50+ members, 2 lake houses in Alabama) but must be architecturally prepared to serve any family group worldwide. The domain is campbellfamilyhub.com.

**Core Philosophy**: This is the digital equivalent of Thanksgiving dinner - warm, welcoming, and centered on preserving what matters most: properties, recipes, photos, music, jokes, and the bonds between generations.

**Micro-interactions to implement**:
- Button hover: subtle lift (translateY -1px) + shadow increase
- Card hover: gentle scale (1.02) + border highlight

## Database Schema

Create ONE comprehensive migration with all tables. Enable RLS on every table. Use simple, effective policies.

### Core Tables

```sql
/*
  # Campbell Family Hub - Complete Database Schema

  ## Overview
  This migration creates the complete database structure for a family
  collaboration platform supporting:
  - Member profiles and family relationships
  - Media library (photos, videos, music)
  - Recipe book with ratings
  - Property management with reservations
  - Polls, voting, and crowdfunding
  - Activity feed and notifications
  - Admin-configurable branding (white-label ready)

  ## Tables Created
  1. hub_settings - Global configuration and branding
  2. user_profiles - Extended member information
  3. family_relationships - Parent/child/sibling connections
  4. invite_codes - Member invitation system
  5. albums - Photo/video collections
  6. media_files - All uploaded media
  7. media_album_items - Album contents
  8. tags - Universal tagging system
  9. media_tags - Media-to-tag relationships
  10. recipes - Family recipe collection
  11. recipe_ratings - Recipe reviews and ratings
  12. jokes - Family humor archive
  13. stories - Family history and memories
  14. properties - Lake houses and shared spaces
  15. reservations - Property booking system
  16. property_projects - Maintenance and improvements
  17. project_updates - Project progress tracking
  18. payments - Financial contribution tracking
  19. polls - Family decision polls
  20. poll_options - Poll choices
  21. poll_votes - Member votes
  22. crowdfunding_campaigns - Family fundraising
  23. campaign_contributions - Crowdfunding pledges
  24. notifications - User notifications
  25. activity_log - Platform-wide activity feed

  ## Security
  - RLS enabled on ALL tables
  - Authenticated users can read most content
  - Users can only modify their own content
  - Admins (is_admin=true) have full access
*/

-- ============================================
-- HUB SETTINGS (White-Label Configuration)
-- ============================================
CREATE TABLE IF NOT EXISTS hub_settings (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  hub_name text NOT NULL DEFAULT 'Family Hub',
  tagline text DEFAULT 'Where family comes together',
  logo_url text,
  favicon_url text,
  family_crest_url text,
  hero_image_url text,
  primary_color text DEFAULT '#c9a962',
  secondary_color text DEFAULT '#7a9a6d',
  default_theme text DEFAULT 'dark' CHECK (default_theme IN ('light', 'dark', 'system')),
  annual_contribution_amount decimal(10,2) DEFAULT 850.00,
  contribution_split_type text DEFAULT 'equal' CHECK (contribution_split_type IN ('equal', 'custom', 'voluntary')),
  enable_reservations boolean DEFAULT true,
  enable_crowdfunding boolean DEFAULT true,
  enable_recipes boolean DEFAULT true,
  enable_media boolean DEFAULT true,
  enable_polls boolean DEFAULT true,
  contact_email text,
  footer_text text,
  custom_css text,
  metadata jsonb DEFAULT '{}',
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- Only one row allowed
CREATE UNIQUE INDEX IF NOT EXISTS hub_settings_singleton ON hub_settings ((true));

-- ============================================
-- USER PROFILES
-- ============================================
CREATE TABLE IF NOT EXISTS user_profiles (
  id uuid PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  email text UNIQUE NOT NULL,
  display_name text NOT NULL,
  avatar_url text DEFAULT 'https://images.unsplash.com/photo-1494790108377-be9c29b29330?w=150&h=150&fit=crop&crop=face',
  cover_photo_url text,
  bio text,
  birthday date,
  phone text,
  location text,
  relationship_type text CHECK (relationship_type IN (
    'patriarch', 'matriarch', 'parent', 'child', 'sibling',
    'grandparent', 'grandchild', 'aunt', 'uncle', 'cousin',
    'in-law', 'spouse', 'friend-of-family', 'other'
  )),
  generation integer, -- 1 = grandparents, 2 = parents, 3 = children, etc.
  is_admin boolean DEFAULT false,
  is_active boolean DEFAULT true,
  theme_preference text DEFAULT 'system' CHECK (theme_preference IN ('light', 'dark', 'system')),
  email_notifications boolean DEFAULT true,
  last_seen_at timestamptz,
  joined_at timestamptz DEFAULT now(),
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- ============================================
-- FAMILY RELATIONSHIPS
-- ============================================
CREATE TABLE IF NOT EXISTS family_relationships (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id uuid NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  related_user_id uuid NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  relationship_type text NOT NULL CHECK (relationship_type IN (
    'parent', 'child', 'sibling', 'spouse', 'grandparent',
    'grandchild', 'aunt', 'uncle', 'cousin', 'in-law'
  )),
  created_at timestamptz DEFAULT now(),
  UNIQUE(user_id, related_user_id)
);

-- ============================================
-- INVITE CODES
-- ============================================
CREATE TABLE IF NOT EXISTS invite_codes (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  code text UNIQUE NOT NULL,
  created_by uuid NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  used_by uuid REFERENCES user_profiles(id) ON DELETE SET NULL,
  email_for text, -- Optional: restrict to specific email
  expires_at timestamptz NOT NULL,
  used_at timestamptz,
  created_at timestamptz DEFAULT now()
);

-- ============================================
-- MEDIA LIBRARY
-- ============================================
CREATE TABLE IF NOT EXISTS albums (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  title text NOT NULL,
  description text,
  cover_image_url text,
  album_type text DEFAULT 'general' CHECK (album_type IN (
    'general', 'event', 'thanksgiving', 'reunion', 'property', 'memories'
  )),
  event_date date,
  is_featured boolean DEFAULT false,
  created_by uuid NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS media_files (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  title text,
  description text,
  file_url text NOT NULL,
  thumbnail_url text,
  media_type text NOT NULL CHECK (media_type IN ('photo', 'video', 'audio')),
  file_size integer,
  mime_type text,
  width integer,
  height integer,
  duration_seconds integer, -- For video/audio
  taken_at timestamptz,
  location_name text,
  latitude decimal(10, 8),
  longitude decimal(11, 8),
  uploaded_by uuid NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  is_featured boolean DEFAULT false,
  view_count integer DEFAULT 0,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS media_album_items (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  album_id uuid NOT NULL REFERENCES albums(id) ON DELETE CASCADE,
  media_id uuid NOT NULL REFERENCES media_files(id) ON DELETE CASCADE,
  sort_order integer DEFAULT 0,
  added_at timestamptz DEFAULT now(),
  UNIQUE(album_id, media_id)
);

CREATE TABLE IF NOT EXISTS tags (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name text UNIQUE NOT NULL,
  tag_type text DEFAULT 'general' CHECK (tag_type IN ('person', 'event', 'location', 'general')),
  linked_user_id uuid REFERENCES user_profiles(id) ON DELETE SET NULL, -- For person tags
  created_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS media_tags (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  media_id uuid NOT NULL REFERENCES media_files(id) ON DELETE CASCADE,
  tag_id uuid NOT NULL REFERENCES tags(id) ON DELETE CASCADE,
  tagged_by uuid REFERENCES user_profiles(id) ON DELETE SET NULL,
  created_at timestamptz DEFAULT now(),
  UNIQUE(media_id, tag_id)
);

-- ============================================
-- RECIPES
-- ============================================
CREATE TABLE IF NOT EXISTS recipes (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  title text NOT NULL,
  description text,
  cover_image_url text,
  category text NOT NULL CHECK (category IN (
    'appetizers', 'main-dishes', 'sides', 'desserts',
    'drinks', 'breakfast', 'thanksgiving-classics', 'other'
  )),
  ingredients jsonb NOT NULL DEFAULT '[]', -- [{item, amount, unit, notes}]
  instructions jsonb NOT NULL DEFAULT '[]', -- [{step, text, image_url}]
  prep_time_minutes integer,
  cook_time_minutes integer,
  servings integer,
  difficulty text DEFAULT 'medium' CHECK (difficulty IN ('easy', 'medium', 'hard')),
  tips text,
  source text, -- "Grandma's recipe", "Aunt Mary", etc.
  is_thanksgiving_classic boolean DEFAULT false,
  is_featured boolean DEFAULT false,
  submitted_by uuid NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  times_made_count integer DEFAULT 0,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS recipe_ratings (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  recipe_id uuid NOT NULL REFERENCES recipes(id) ON DELETE CASCADE,
  user_id uuid NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  rating integer NOT NULL CHECK (rating >= 1 AND rating <= 5),
  comment text,
  made_it boolean DEFAULT false,
  photo_url text, -- "Here's how mine turned out!"
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now(),
  UNIQUE(recipe_id, user_id)
);

-- ============================================
-- JOKES & STORIES
-- ============================================
CREATE TABLE IF NOT EXISTS jokes (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  content text NOT NULL,
  punchline text,
  category text DEFAULT 'general' CHECK (category IN (
    'general', 'dad-jokes', 'family-moments', 'inside-jokes', 'kids-say'
  )),
  submitted_by uuid NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  upvote_count integer DEFAULT 0,
  created_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS joke_votes (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  joke_id uuid NOT NULL REFERENCES jokes(id) ON DELETE CASCADE,
  user_id uuid NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  vote_type text NOT NULL CHECK (vote_type IN ('up', 'down')),
  created_at timestamptz DEFAULT now(),
  UNIQUE(joke_id, user_id)
);

CREATE TABLE IF NOT EXISTS stories (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  title text NOT NULL,
  content text NOT NULL,
  story_date date,
  decade text, -- "1980s", "2000s", etc.
  cover_image_url text,
  related_media_ids uuid[] DEFAULT '{}',
  submitted_by uuid NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  is_featured boolean DEFAULT false,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- ============================================
-- PROPERTIES
-- ============================================
CREATE TABLE IF NOT EXISTS properties (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name text NOT NULL,
  description text,
  address text,
  city text,
  state text,
  zip_code text,
  latitude decimal(10, 8),
  longitude decimal(11, 8),
  cover_image_url text,
  photo_urls text[] DEFAULT '{}',
  amenities text[] DEFAULT '{}',
  house_rules text,
  max_guests integer,
  bedrooms integer,
  bathrooms decimal(3,1),
  property_type text DEFAULT 'lake-house' CHECK (property_type IN (
    'lake-house', 'cabin', 'beach-house', 'farm', 'city-home', 'other'
  )),
  is_active boolean DEFAULT true,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS reservations (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  property_id uuid NOT NULL REFERENCES properties(id) ON DELETE CASCADE,
  user_id uuid NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  start_date date NOT NULL,
  end_date date NOT NULL,
  guest_count integer NOT NULL,
  guests_names text, -- List of who's coming
  purpose text, -- "Weekend getaway", "Family reunion", etc.
  notes text,
  status text DEFAULT 'pending' CHECK (status IN (
    'pending', 'approved', 'denied', 'cancelled'
  )),
  admin_notes text,
  reviewed_by uuid REFERENCES user_profiles(id) ON DELETE SET NULL,
  reviewed_at timestamptz,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now(),
  CONSTRAINT valid_date_range CHECK (end_date > start_date)
);

CREATE TABLE IF NOT EXISTS property_blackout_dates (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  property_id uuid NOT NULL REFERENCES properties(id) ON DELETE CASCADE,
  start_date date NOT NULL,
  end_date date NOT NULL,
  reason text,
  created_by uuid REFERENCES user_profiles(id) ON DELETE SET NULL,
  created_at timestamptz DEFAULT now()
);

-- ============================================
-- PROJECTS
-- ============================================
CREATE TABLE IF NOT EXISTS property_projects (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  property_id uuid NOT NULL REFERENCES properties(id) ON DELETE CASCADE,

-- Auto-create user profile on auth signup
CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO user_profiles (id, email, display_name)
  VALUES (
    NEW.id,
    NEW.email,
    COALESCE(NEW.raw_user_meta_data->>'display_name', split_part(NEW.email, '@', 1))
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();

-- Update campaign totals when contribution is made
CREATE OR REPLACE FUNCTION update_campaign_total()
RETURNS TRIGGER AS $$
BEGIN
  UPDATE crowdfunding_campaigns
  SET current_amount = (
    SELECT COALESCE(SUM(amount), 0)
    FROM campaign_contributions
    WHERE campaign_id = NEW.campaign_id
  )
  WHERE id = NEW.campaign_id;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_campaign_on_contribution
  AFTER INSERT ON campaign_contributions
  FOR EACH ROW EXECUTE FUNCTION update_campaign_total();

-- Update joke upvote count
CREATE OR REPLACE FUNCTION update_joke_votes()
RETURNS TRIGGER AS $$
BEGIN
  IF TG_OP = 'INSERT' OR TG_OP = 'UPDATE' THEN
    UPDATE jokes SET upvote_count = (
      SELECT COUNT(*) FROM joke_votes WHERE joke_id = NEW.joke_id AND vote_type = 'up'
    ) - (
      SELECT COUNT(*) FROM joke_votes WHERE joke_id = NEW.joke_id AND vote_type = 'down'
    ) WHERE id = NEW.joke_id;
    RETURN NEW;
  ELSIF TG_OP = 'DELETE' THEN
    UPDATE jokes SET upvote_count = (
      SELECT COUNT(*) FROM joke_votes WHERE joke_id = OLD.joke_id AND vote_type = 'up'
    ) - (
      SELECT COUNT(*) FROM joke_votes WHERE joke_id = OLD.joke_id AND vote_type = 'down'
    ) WHERE id = OLD.joke_id;
    RETURN OLD;
  END IF;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_joke_votes_trigger
  AFTER INSERT OR UPDATE OR DELETE ON joke_votes
  FOR EACH ROW EXECUTE FUNCTION update_joke_votes();
```

## Storage Configuration

**IMPORTANT**: Storage buckets must be created MANUALLY via the Supabase Dashboard (Storage section). This cannot be done reliably via SQL in all environments. Create these 3 public buckets:

| Bucket Name | Purpose | Public |
|------------|---------|--------|
| `family-media` | Photos, videos, audio files | Yes |
| `avatars` | Profile pictures | Yes |
| `branding` | Hub logos, hero images, branding assets | Yes |

**To create buckets in Supabase Dashboard:**
1. Go to Storage in the left sidebar
2. Click "Create a new bucket"
3. Enter the bucket name exactly as shown above
4. Toggle "Public bucket" to ON
5. Click "Save"
6. Repeat for each bucket

After creating buckets, apply these storage policies via SQL Editor:

```sql
-- Storage policies (run AFTER creating buckets manually)
CREATE POLICY "Authenticated users can upload media" ON storage.objects FOR INSERT TO authenticated WITH CHECK (bucket_id IN ('family-media', 'avatars'));
CREATE POLICY "Authenticated users can view media" ON storage.objects FOR SELECT TO authenticated USING (bucket_id IN ('family-media', 'avatars', 'branding'));
CREATE POLICY "Users can update own media" ON storage.objects FOR UPDATE TO authenticated USING (auth.uid()::text = (storage.foldername(name))[1]);
CREATE POLICY "Users can delete own media" ON storage.objects FOR DELETE TO authenticated USING (auth.uid()::text = (storage.foldername(name))[1]);
CREATE POLICY "Admins can manage branding" ON storage.objects FOR ALL TO authenticated USING (bucket_id = 'branding');
CREATE POLICY "Public can view public media" ON storage.objects FOR SELECT USING (bucket_id IN ('family-media', 'avatars', 'branding'));
```

## Key User Flows

### New Member Onboarding

1. Admin generates invite code from Admin > User Management
2. Admin shares invite link with new member (email, text, etc.)
3. New member clicks link, taken to /register?code=XXXXX
4. System validates code
5. Member completes registration form
6. Account created, code marked used
7. Member lands on home page with welcome message
8. First-time tooltip tour highlights key features

### Making a Reservation

1. Member visits Properties page
2. Clicks on desired property
3. Views calendar for availability
4. Clicks "Request Reservation"
5. Fills form: dates, guests, purpose
6. Submits request
7. System validates no conflicts
8. Creates pending reservation
9. Admins receive notification
10. Admin approves/denies
11. Member receives notification of decision
12. Approved reservation appears on calendar

### Contributing to Campaign

1. Member sees campaign on home or crowdfunding page
2. Clicks "Contribute"
3. Modal opens with amount input
4. Enters amount, optional message
5. Toggles anonymous if desired
6. Confirms contribution
7. Contribution recorded
8. Campaign total updated in real-time
9. Activity logged
10. If campaign reaches goal: celebrate!

### Uploading Photos

1. Member clicks Upload button (FAB or header)
2. Drag-drop or select files
3. Preview appears for each file
4. Adds title, description, tags
5. Submits
6. Progress bar shows upload
7. On complete: redirected to gallery or success message
8. Photos appear in gallery
9. Activity logged for family to see

## Error Handling

### Error States

All error states should:
- Use warm, friendly language
- Include helpful suggestions
- Match the family-friendly tone

**Examples**:
- 404: "Looks like this page wandered off! Let's get you back home."
- Network error: "We're having trouble connecting. Check your internet and try again."
- Form validation: Inline messages below fields, red but not harsh

### Loading States

- Use skeleton screens, not spinners
- Match the layout of actual content
- Subtle pulse animation

## Final Notes

This specification represents a complete family collaboration platform built with modern best practices. The architecture is intentionally simple - flat permission model, direct database operations, clear component boundaries.

**Key differences from complex enterprise patterns**:
- Single boolean `is_admin` instead of role tables
- RLS handles all security
- No edge functions needed for basic operations
- Single comprehensive migration instead of iterative
- White-label ready through `hub_settings` table

**Build with love**. This is a family gathering place. Every interaction should feel warm, every feature should strengthen family bonds. The best technology is invisible - it just brings people together.

```typescript
import { createContext, useContext, useEffect, useState, ReactNode } from 'react';
import { User, Session } from '@supabase/supabase-js';
import { supabase } from '../lib/supabase';
import { UserProfile } from '../types';

interface AuthContextType {
  user: User | null;
  profile: UserProfile | null;
  session: Session | null;
  loading: boolean;
  signIn: (email: string, password: string) => Promise<{ error: Error | null }>;
  signUp: (email: string, password: string, displayName: string) => Promise<{ error: Error | null }>;
  signOut: () => Promise<void>;
  refreshProfile: () => Promise<void>;
}

const AuthContext = createContext<AuthContextType | undefined>(undefined);

export function AuthProvider({ children }: { children: ReactNode }) {
  const [user, setUser] = useState<User | null>(null);
  const [profile, setProfile] = useState<UserProfile | null>(null);
  const [session, setSession] = useState<Session | null>(null);
  const [loading, setLoading] = useState(true);

  const fetchProfile = async (userId: string) => {
    const { data } = await supabase
      .from('user_profiles')
      .select('*')
      .eq('id', userId)
      .maybeSingle();
    setProfile(data);
  };

  useEffect(() => {
    supabase.auth.getSession().then(({ data: { session } }) => {
      setSession(session);
      setUser(session?.user ?? null);
      if (session?.user) {
        fetchProfile(session.user.id);
      }
      setLoading(false);
    });

    const { data: { subscription } } = supabase.auth.onAuthStateChange((_event, session) => {
      setSession(session);
      setUser(session?.user ?? null);
      if (session?.user) {
        fetchProfile(session.user.id);
      } else {
        setProfile(null);
      }
    });

    return () => subscription.unsubscribe();
  }, []);

  const signIn = async (email: string, password: string) => {
    const { error } = await supabase.auth.signInWithPassword({ email, password });
    return { error };
  };

  const signUp = async (email: string, password: string, displayName: string) => {
    const { error } = await supabase.auth.signUp({
      email,
      password,
      options: { data: { display_name: displayName } },
    });
    return { error };
  };

  const signOut = async () => {
    await supabase.auth.signOut();
  };

  const refreshProfile = async () => {
    if (user) {
      await fetchProfile(user.id);
    }
  };

  return (
    <AuthContext.Provider value={{ user, profile, session, loading, signIn, signUp, signOut, refreshProfile }}>
      {children}
    </AuthContext.Provider>
  );
}

export function useAuth() {
  const context = useContext(AuthContext);
  if (!context) {
    throw new Error('useAuth must be used within an AuthProvider');
  }
  return context;
}
```

### Theme Context (`src/contexts/ThemeContext.tsx`)

```typescript
import { createContext, useContext, useEffect, useState, ReactNode } from 'react';

type Theme = 'light' | 'dark' | 'system';

interface ThemeContextType {
  theme: Theme;
  resolvedTheme: 'light' | 'dark';
  setTheme: (theme: Theme) => void;
}

const ThemeContext = createContext<ThemeContextType | undefined>(undefined);

export function ThemeProvider({ children }: { children: ReactNode }) {
  const [theme, setThemeState] = useState<Theme>(() => {
    const stored = localStorage.getItem('theme');
    return (stored as Theme) || 'dark';
  });

  const [resolvedTheme, setResolvedTheme] = useState<'light' | 'dark'>('dark');

  useEffect(() => {
    const root = document.documentElement;

    const resolve = () => {
      if (theme === 'system') {
        return window.matchMedia('(prefers-color-scheme: light)').matches ? 'light' : 'dark';
      }
      return theme;
    };

    const resolved = resolve();
    setResolvedTheme(resolved);

    root.classList.remove('light', 'dark');
    root.classList.add(resolved);
    localStorage.setItem('theme', theme);
  }, [theme]);

  useEffect(() => {
    if (theme === 'system') {
      const mediaQuery = window.matchMedia('(prefers-color-scheme: light)');
      const handler = () => setResolvedTheme(mediaQuery.matches ? 'light' : 'dark');
      mediaQuery.addEventListener('change', handler);
      return () => mediaQuery.removeEventListener('change', handler);
    }
  }, [theme]);

  const setTheme = (newTheme: Theme) => {
    setThemeState(newTheme);
  };

  return (
    <ThemeContext.Provider value={{ theme, resolvedTheme, setTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

export function useTheme() {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within a ThemeProvider');
  }
  return context;
}
```

## Appendix G: Admin Bootstrap Edge Function

Deploy this edge function to create the first admin user:

**Function name**: `bootstrap-admin`

```typescript
import "jsr:@supabase/functions-js/edge-runtime.d.ts";
import { createClient } from "npm:@supabase/supabase-js@2";

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Methods": "POST, OPTIONS",
  "Access-Control-Allow-Headers": "Content-Type, Authorization, X-Client-Info, Apikey",
};

Deno.serve(async (req: Request) => {
  if (req.method === "OPTIONS") {
    return new Response(null, { status: 200, headers: corsHeaders });
  }

  try {
    const { email, password, displayName, bootstrapSecret } = await req.json();

    // Simple secret check - set this in your edge function secrets
    const expectedSecret = Deno.env.get("BOOTSTRAP_SECRET");
    if (!expectedSecret || bootstrapSecret !== expectedSecret) {
      return new Response(
        JSON.stringify({ error: "Invalid bootstrap secret" }),
        { status: 401, headers: { ...corsHeaders, "Content-Type": "application/json" } }
      );
    }

    const supabaseAdmin = createClient(
      Deno.env.get("SUPABASE_URL")!,
      Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!
    );

    // Check if any admin exists
    const { data: existingAdmins } = await supabaseAdmin
      .from("user_profiles")
      .select("id")
      .eq("is_admin", true)
      .limit(1);

    if (existingAdmins && existingAdmins.length > 0) {
      return new Response(
        JSON.stringify({ error: "Admin already exists. Use the app to invite more admins." }),
        { status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" } }
      );
    }

    // Create auth user
    const { data: authData, error: authError } = await supabaseAdmin.auth.admin.createUser({
      email,
      password,
      email_confirm: true,
      user_metadata: { display_name: displayName },
    });

    if (authError) {
      return new Response(
        JSON.stringify({ error: authError.message }),
        { status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" } }
      );
    }

    // Update profile to be admin
    const { error: profileError } = await supabaseAdmin
      .from("user_profiles")
      .update({ is_admin: true, display_name: displayName })
      .eq("id", authData.user.id);

    if (profileError) {
      return new Response(
        JSON.stringify({ error: profileError.message }),
        { status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" } }
      );
    }

    return new Response(
      JSON.stringify({ success: true, message: "Admin user created successfully" }),
      { headers: { ...corsHeaders, "Content-Type": "application/json" } }
    );
  } catch (error) {
    return new Response(
      JSON.stringify({ error: error.message }),
      { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } }
    );
  }
```

## Appendix M: Toast System

### `src/contexts/ToastContext.tsx`

```typescript
import { createContext, useContext, useState, useCallback, ReactNode } from 'react';
import { Toast, ToastContainer } from '../components/ui/Toast';

interface ToastItem {
  id: string;
  type: 'success' | 'error' | 'warning' | 'info';
  message: string;
}

interface ToastContextType {
  toast: (type: ToastItem['type'], message: string) => void;
  success: (message: string) => void;
  error: (message: string) => void;
  warning: (message: string) => void;
  info: (message: string) => void;
}

const ToastContext = createContext<ToastContextType | undefined>(undefined);

export function ToastProvider({ children }: { children: ReactNode }) {
  const [toasts, setToasts] = useState<ToastItem[]>([]);

  const removeToast = useCallback((id: string) => {
    setToasts((prev) => prev.filter((t) => t.id !== id));
  }, []);

  const toast = useCallback((type: ToastItem['type'], message: string) => {
    const id = crypto.randomUUID();
    setToasts((prev) => [...prev, { id, type, message }]);
    setTimeout(() => removeToast(id), 5000);
  }, [removeToast]);

  const success = useCallback((message: string) => toast('success', message), [toast]);
  const error = useCallback((message: string) => toast('error', message), [toast]);
  const warning = useCallback((message: string) => toast('warning', message), [toast]);
  const info = useCallback((message: string) => toast('info', message), [toast]);

  return (
    <ToastContext.Provider value={{ toast, success, error, warning, info }}>
      {children}
      <ToastContainer toasts={toasts} onClose={removeToast} />
    </ToastContext.Provider>
  );
}

export function useToast() {
  const context = useContext(ToastContext);
  if (!context) {
    throw new Error('useToast must be used within a ToastProvider');
  }
  return context;
}
```

## Appendix N: Hub Context

### `src/contexts/HubContext.tsx`

```typescript
import { createContext, useContext, useEffect, useState, ReactNode } from 'react';
import { supabase } from '../lib/supabase';
import { HubSettings } from '../types';

interface HubContextType {
  settings: HubSettings | null;
  loading: boolean;
  refreshSettings: () => Promise<void>;
}

const HubContext = createContext<HubContextType | undefined>(undefined);

export function HubProvider({ children }: { children: ReactNode }) {
  const [settings, setSettings] = useState<HubSettings | null>(null);
  const [loading, setLoading] = useState(true);

  const fetchSettings = async () => {
    const { data } = await supabase
      .from('hub_settings')
      .select('*')
      .limit(1)
      .maybeSingle();

    setSettings(data);
    setLoading(false);
  };

  useEffect(() => {
    fetchSettings();
  }, []);

  const refreshSettings = async () => {
    await fetchSettings();
  };

  return (
    <HubContext.Provider value={{ settings, loading, refreshSettings }}>
      {children}
    </HubContext.Provider>
  );
}

export function useHub() {
  const context = useContext(HubContext);
  if (!context) {
    throw new Error('useHub must be used within a HubProvider');
  }
  return context;
}
```

```typescript
    if (isOpen) {

    } else {
      document.body.style.overflow = 'unset';

  const sizes = {
    sm: 'max-w-sm',
    md: 'max-w-md',
    lg: 'max-w-lg',
    xl: 'max-w-2xl',
  };

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center p-4">
      <div

        onClick={onClose}
```

```typescript
type ClassValue = string | undefined | null | false;

export function cn(...inputs: ClassValue[]) {
  return inputs.filter(Boolean).join(' ');
}

export function formatDate(date: string | Date, options?: Intl.DateTimeFormatOptions): string {
  const d = typeof date === 'string' ? new Date(date) : date;
  return d.toLocaleDateString('en-US', {
    year: 'numeric',
    month: 'long',
    day: 'numeric',
    ...options,
  });
}

export function formatRelativeTime(date: string | Date): string {
  const d = typeof date === 'string' ? new Date(date) : date;
  const now = new Date();
  const diffMs = now.getTime() - d.getTime();
  const diffMins = Math.floor(diffMs / 60000);
  const diffHours = Math.floor(diffMins / 60);
  const diffDays = Math.floor(diffHours / 24);

  if (diffMins < 1) return 'just now';
  if (diffMins < 60) return `${diffMins}m ago`;
  if (diffHours < 24) return `${diffHours}h ago`;
  if (diffDays < 7) return `${diffDays}d ago`;
  return formatDate(d, { month: 'short', day: 'numeric' });
}

export function formatCurrency(amount: number): string {
  return new Intl.NumberFormat('en-US', {
    style: 'currency',
    currency: 'USD',
    minimumFractionDigits: 0,
    maximumFractionDigits: 0,
  }).format(amount);
}

export function generateInviteCode(): string {
  const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';
  let code = '';
  for (let i = 0; i < 8; i++) {
    code += chars.charAt(Math.floor(Math.random() * chars.length));
  }
  return code;
}

export function slugify(text: string): string {
  return text
    .toLowerCase()
    .replace(/[^\w\s-]/g, '')
    .replace(/[\s_-]+/g, '-')
    .replace(/^-+|-+$/g, '');
}

export function truncate(text: string, length: number): string {
  if (text.length <= length) return text;
  return text.slice(0, length).trim() + '...';
}

export function getInitials(name: string): string {
  return name
    .split(' ')
    .map((word) => word[0])
    .join('')
    .toUpperCase()
    .slice(0, 2);
}
```

```typescript
import { supabase } from './supabase';

export type StorageBucket = 'family-media' | 'avatars' | 'branding';

interface UploadOptions {
  bucket: StorageBucket;
  path: string;
  file: File;
  onProgress?: (progress: number) => void;
}

interface UploadResult {
  url: string | null;
  error: string | null;
}

export async function uploadFile({ bucket, path, file }: UploadOptions): Promise<UploadResult> {
  const fileExt = file.name.split('.').pop();
  const fileName = `${Date.now()}-${Math.random().toString(36).substring(2)}.${fileExt}`;
  const filePath = path ? `${path}/${fileName}` : fileName;

  const { error } = await supabase.storage.from(bucket).upload(filePath, file, {
    cacheControl: '3600',
    upsert: false,
  });

  if (error) {
    return { url: null, error: error.message };
  }

  const { data: urlData } = supabase.storage.from(bucket).getPublicUrl(filePath);

  return { url: urlData.publicUrl, error: null };
}

export async function deleteFile(bucket: StorageBucket, path: string): Promise<{ error: string | null }> {
  const { error } = await supabase.storage.from(bucket).remove([path]);

  if (error) {
    return { error: error.message };
  }

  return { error: null };
}

export function getPublicUrl(bucket: StorageBucket, path: string): string {
  const { data } = supabase.storage.from(bucket).getPublicUrl(path);
  return data.publicUrl;
}

export function validateImageFile(file: File): { valid: boolean; error?: string } {
  const validTypes = ['image/jpeg', 'image/png', 'image/gif', 'image/webp'];
  const maxSize = 10 * 1024 * 1024;

  if (!validTypes.includes(file.type)) {
    return { valid: false, error: 'Please upload a JPEG, PNG, GIF, or WebP image.' };
  }

  if (file.size > maxSize) {
    return { valid: false, error: 'Image must be under 10MB.' };
  }

  return { valid: true };
}

export function validateVideoFile(file: File): { valid: boolean; error?: string } {
  const validTypes = ['video/mp4', 'video/webm', 'video/quicktime'];
  const maxSize = 100 * 1024 * 1024;

  if (!validTypes.includes(file.type)) {
    return { valid: false, error: 'Please upload an MP4, WebM, or MOV video.' };
  }

  if (file.size > maxSize) {
    return { valid: false, error: 'Video must be under 100MB.' };
  }
```

```typescript
import { Component, ErrorInfo, ReactNode } from 'react';
import { Button } from './ui/Button';

interface Props {
  children: ReactNode;
}

interface State {
  hasError: boolean;
  error?: Error;
}

export class ErrorBoundary extends Component<Props, State> {
  public state: State = { hasError: false };

  public static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  public componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    console.error('Uncaught error:', error, errorInfo);
  }

  public render() {
    if (this.state.hasError) {
      return (
        <div className="min-h-screen bg-[var(--bg-primary)] flex items-center justify-center p-4">
          <div className="text-center max-w-md">
            <div className="h-20 w-20 mx-auto mb-6 rounded-full bg-error/20 flex items-center justify-center">
              <span className="text-4xl">:(</span>

            <p className="text-[var(--text-secondary)] mb-6">
              We're sorry, but something unexpected happened. Please try refreshing the page.
            </p>
            <Button onClick={() => window.location.reload()}>
              Refresh Page
            </Button>
          </div>
        </div>
      );
    }

    return this.props.children;
  }
}
```

### `src/pages/media/Upload.tsx`

```typescript
import { useState, useRef } from 'react';
import { useNavigate } from 'react-router-dom';
import { supabase } from '../../lib/supabase';
import { useAuth } from '../../contexts/AuthContext';
import { useToast } from '../../contexts/ToastContext';
import { Card } from '../../components/ui/Card';
import { Button } from '../../components/ui/Button';
import { Input } from '../../components/ui/Input';
import { Upload as UploadIcon, X } from 'lucide-react';

export default function MediaUpload() {
  const { user } = useAuth();
  const { success, error: showError } = useToast();
  const navigate = useNavigate();
  const fileInputRef = useRef<HTMLInputElement>(null);
  const [files, setFiles] = useState<File[]>([]);
  const [uploading, setUploading] = useState(false);
  const [title, setTitle] = useState('');

  const handleFiles = (e: React.ChangeEvent<HTMLInputElement>) => {
    if (e.target.files) setFiles(Array.from(e.target.files));
  };

  const handleUpload = async () => {
    if (!files.length || !user) return;
    setUploading(true);
    try {
      for (const file of files) {
        const ext = file.name.split('.').pop();
        const path = `${user.id}/${Date.now()}.${ext}`;
        const { error: uploadError } = await supabase.storage.from('family-media').upload(path, file);
        if (uploadError) throw uploadError;
        const { data: { publicUrl } } = supabase.storage.from('family-media').getPublicUrl(path);
        await supabase.from('media_files').insert({ file_url: publicUrl, title: title || file.name, media_type: file.type.startsWith('video') ? 'video' : file.type.startsWith('audio') ? 'audio' : 'photo', uploaded_by: user.id, mime_type: file.type, file_size: file.size });
      }
      success('Upload complete!');
      navigate('/media');
    } catch (err: any) { showError(err.message); }
    setUploading(false);
  };

  return (
    <div className="max-w-2xl mx-auto px-4 py-8">
      <h1 className="font-serif text-3xl text-[var(--text-primary)] mb-8">Upload Media</h1>
      <Card>
        <div className="border-2 border-dashed border-[var(--border-default)] rounded-lg p-8 text-center cursor-pointer hover:border-accent-gold transition-colors" onClick={() => fileInputRef.current?.click()}>
          <UploadIcon size={48} className="mx-auto text-[var(--text-muted)] mb-4" />
          <p className="text-[var(--text-secondary)] mb-2">Drag and drop or click to select</p>
          <p className="text-sm text-[var(--text-muted)]">Photos, videos, or audio files</p>
          <input ref={fileInputRef} type="file" multiple accept="image/*,video/*,audio/*" onChange={handleFiles} className="hidden" />
        </div>
        {files.length > 0 && (
          <div className="mt-4 space-y-2">
            {files.map((f, i) => (
              <div key={i} className="flex items-center justify-between p-2 bg-[var(--bg-secondary)] rounded">
                <span className="text-sm text-[var(--text-primary)] truncate">{f.name}</span>
                <button onClick={() => setFiles(files.filter((_, j) => j !== i))} className="text-[var(--text-muted)] hover:text-error"><X size={16} /></button>
              </div>
            ))}
          </div>
        )}
        <Input label="Title (optional)" value={title} onChange={e => setTitle(e.target.value)} className="mt-4" placeholder="Give your upload a name" />
        <Button onClick={handleUpload} loading={uploading} disabled={!files.length} className="w-full mt-4">Upload {files.length > 0 && `(${files.length} file${files.length > 1 ? 's' : ''})`}</Button>
      </Card>
    </div>
  );
}
```

```typescript
import { Button } from '../../components/ui/Button';
import { Skeleton } from '../../components/ui/Skeleton';
import { ArrowLeft, Check } from 'lucide-react';

export default function PollDetail() {
  const { id } = useParams();
  const { user } = useAuth();
  const { success, error: showError } = useToast();
  const [poll, setPoll] = useState<any>(null);
  const [options, setOptions] = useState<any[]>([]);
  const [votes, setVotes] = useState<any[]>([]);
  const [userVote, setUserVote] = useState<string | null>(null);
  const [loading, setLoading] = useState(true);
  const [voting, setVoting] = useState(false);

  useEffect(() => {
    const fetch = async () => {
      const { data: pollData } = await supabase.from('polls').select('*').eq('id', id).maybeSingle();
      const { data: optionsData } = await supabase.from('poll_options').select('*').eq('poll_id', id).order('sort_order');
      const { data: votesData } = await supabase.from('poll_votes').select('option_id, user_id').eq('poll_id', id);
      setPoll(pollData);
      setOptions(optionsData || []);
      setVotes(votesData || []);
      if (user) setUserVote(votesData?.find(v => v.user_id === user.id)?.option_id || null);
      setLoading(false);
    };
    if (id) fetch();
  }, [id, user]);

  const handleVote = async (optionId: string) => {
    if (!user || !poll) return;
    setVoting(true);
    if (userVote) await supabase.from('poll_votes').delete().eq('poll_id', id).eq('user_id', user.id);
    const { error } = await supabase.from('poll_votes').insert({ poll_id: id, option_id: optionId, user_id: user.id });
    if (error) showError(error.message);
    else { setUserVote(optionId); setVotes([...votes.filter(v => v.user_id !== user.id), { option_id: optionId, user_id: user.id }]); success('Vote recorded!'); }
    setVoting(false);
  };

  if (loading) return <div className="max-w-2xl mx-auto px-4 py-8"><Skeleton className="h-64 w-full" /></div>;
  if (!poll) return <div className="max-w-2xl mx-auto px-4 py-8"><p className="text-[var(--text-muted)]">Poll not found</p></div>;

  const isOpen = new Date(poll.closes_at) > new Date();
  const getVoteCount = (optionId: string) => votes.filter(v => v.option_id === optionId).length;
  const totalVotes = votes.length;

  return (
    <div className="max-w-2xl mx-auto px-4 py-8">
      <Link to="/polls" className="inline-flex items-center gap-2 text-[var(--text-secondary)] hover:text-[var(--text-primary)] mb-6"><ArrowLeft size={18} /> Back to Polls</Link>

      {poll.description && <p className="text-[var(--text-secondary)] mb-6">{poll.description}</p>}
      <Card>
        <div className="space-y-3">
          {options.map(opt => {
            const count = getVoteCount(opt.id);
            const pct = totalVotes > 0 ? Math.round((count / totalVotes) * 100) : 0;
            const isSelected = userVote === opt.id;
            return (
              <button key={opt.id} onClick={() => isOpen && handleVote(opt.id)} disabled={!isOpen || voting} className={`w-full text-left p-4 rounded-lg border transition-all ${isSelected ? 'border-accent-gold bg-accent-gold/10' : 'border-[var(--border-default)] hover:border-[var(--border-emphasis)]'}`}>
                <div className="flex items-center justify-between mb-2">
                  <span className="text-[var(--text-primary)] flex items-center gap-2">{isSelected && <Check size={16} className="text-accent-gold" />}{opt.option_text}</span>
                  <span className="text-sm text-[var(--text-muted)]">{pct}%</span>
                </div>
                <div className="h-2 bg-[var(--bg-secondary)] rounded-full overflow-hidden"><div className="h-full bg-accent-gold rounded-full transition-all" style={{ width: `${pct}%` }} /></div>
              </button>
            );
          })}
        </div>
        <p className="text-sm text-[var(--text-muted)] mt-4">{totalVotes} vote{totalVotes !== 1 ? 's' : ''} - {isOpen ? 'Poll is open' : 'Poll is closed'}</p>
      </Card>
    </div>
  );
}
```

### `src/pages/jokes/JokeList.tsx`

```typescript
import { useEffect, useState } from 'react';
import { supabase } from '../../lib/supabase';
import { useAuth } from '../../contexts/AuthContext';
import { Card } from '../../components/ui/Card';
import { CardSkeleton } from '../../components/ui/Skeleton';
import { ThumbsUp, ThumbsDown, ChevronDown, ChevronUp } from 'lucide-react';

interface Joke { id: string; content: string; punchline: string | null; upvote_count: number; }

export default function JokeList() {
  const { user } = useAuth();
  const [jokes, setJokes] = useState<Joke[]>([]);
  const [revealed, setRevealed] = useState<Set<string>>(new Set());
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const fetch = async () => {
      const { data } = await supabase.from('jokes').select('id, content, punchline, upvote_count').order('upvote_count', { ascending: false });
      setJokes(data || []);
      setLoading(false);
    };
    fetch();
  }, []);

  const toggleReveal = (id: string) => setRevealed(prev => { const n = new Set(prev); n.has(id) ? n.delete(id) : n.add(id); return n; });

  const vote = async (jokeId: string, type: 'up' | 'down') => {
    if (!user) return;
    await supabase.from('joke_votes').upsert({ joke_id: jokeId, user_id: user.id, vote_type: type }, { onConflict: 'joke_id,user_id' });
    const { data } = await supabase.from('jokes').select('id, upvote_count').eq('id', jokeId).maybeSingle();
    if (data) setJokes(jokes.map(j => j.id === jokeId ? { ...j, upvote_count: data.upvote_count } : j));
  };

  return (
    <div className="max-w-2xl mx-auto px-4 py-8">
      <h1 className="font-serif text-3xl text-[var(--text-primary)] mb-8">Family Jokes</h1>
      {loading ? (
        <div className="space-y-4">{[1,2,3,4,5].map(i => <CardSkeleton key={i} />)}</div>
      ) : jokes.length > 0 ? (
        <div className="space-y-4">
          {jokes.map(j => (
            <Card key={j.id}>
              <p className="text-[var(--text-primary)] text-lg">{j.content}</p>
              {j.punchline && (
                <div className="mt-3">
                  {revealed.has(j.id) ? (
                    <p className="text-accent-gold font-medium">{j.punchline}</p>
                  ) : (
                    <button onClick={() => toggleReveal(j.id)} className="text-sm text-accent-gold flex items-center gap-1"><ChevronDown size={16} />Reveal punchline</button>
                  )}
                </div>
              )}
```

```typescript
interface User { id: string; email: string; display_name: string; avatar_url: string | null; is_admin: boolean; is_active: boolean; }

export default function AdminUserManagement() {
  const { success, error: showError } = useToast();
  const [users, setUsers] = useState<User[]>([]);
  const [loading, setLoading] = useState(true);
  const [inviteEmail, setInviteEmail] = useState('');
  const [inviteLink, setInviteLink] = useState('');

  useEffect(() => {
    const fetch = async () => {
      const { data } = await supabase.from('user_profiles').select('id, email, display_name, avatar_url, is_admin, is_active').order('display_name');
      setUsers(data || []);
      setLoading(false);
    };
    fetch();
  }, []);

  const generateInvite = async () => {
    const code = generateInviteCode();
    const expires = new Date(); expires.setDate(expires.getDate() + 7);
    const { data: { user } } = await supabase.auth.getUser();
    if (!user) return;
    const { error } = await supabase.from('invite_codes').insert({ code, created_by: user.id, email_for: inviteEmail || null, expires_at: expires.toISOString() });
    if (error) showError(error.message);
    else { setInviteLink(`${window.location.origin}/register?code=${code}`); success('Invite created!'); }
  };

  const copyLink = () => { navigator.clipboard.writeText(inviteLink); success('Copied!'); };

  const toggleAdmin = async (userId: string, isAdmin: boolean) => {
    const { error } = await supabase.from('user_profiles').update({ is_admin: !isAdmin }).eq('id', userId);
    if (error) showError(error.message);
    else { setUsers(users.map(u => u.id === userId ? { ...u, is_admin: !isAdmin } : u)); success('Updated!'); }
  };

  return (
    <div className="max-w-4xl mx-auto px-4 py-8">
      <h1 className="font-serif text-3xl text-[var(--text-primary)] mb-8">User Management</h1>
      <Card className="mb-8">
        <h2 className="font-medium text-[var(--text-primary)] mb-4 flex items-center gap-2"><UserPlus size={20} />Invite New Member</h2>
        <div className="flex gap-2">
          <Input placeholder="Email (optional)" value={inviteEmail} onChange={e => setInviteEmail(e.target.value)} />
          <Button onClick={generateInvite}>Generate</Button>
        </div>
        {inviteLink && (
          <div className="mt-4 flex gap-2">
            <Input value={inviteLink} readOnly />
            <Button variant="secondary" onClick={copyLink}><Copy size={16} /></Button>
          </div>
        )}
      </Card>
      <h2 className="font-medium text-[var(--text-primary)] mb-4">Members ({users.length})</h2>
      {loading ? (
        <div className="space-y-2">{[1,2,3,4,5].map(i => <CardSkeleton key={i} />)}</div>
      ) : (
        <div className="space-y-2">
          {users.map(u => (
            <Card key={u.id} hover={false} className="flex items-center gap-4 p-4">
              <img src={u.avatar_url || 'https://images.pexels.com/photos/1222271/pexels-photo-1222271.jpeg?w=150'} alt="" className="h-10 w-10 rounded-full object-cover" />
              <div className="flex-1">

                <p className="text-sm text-[var(--text-muted)]">{u.email}</p>
              </div>
              <Button size="sm" variant={u.is_admin ? 'secondary' : 'ghost'} onClick={() => toggleAdmin(u.id, u.is_admin)}>{u.is_admin ? 'Remove Admin' : 'Make Admin'}</Button>
            </Card>
```
